\begin{frame}
  \frametitle{State of knowledge}

  \begin{block}{Preliminary conclusions}
    Little evidence for age-independent pattern.

    Occupancy has largest effect. 
    
    P(carbonate \(\mid\) occurrences) \(\approx\) no effect.
  \end{block}
\end{frame}
